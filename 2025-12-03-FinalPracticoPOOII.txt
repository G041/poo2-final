function Trailer(tipo, tiposDeCarga) {
    this.tipo = tipo;
    this.tiposDeCarga = Array.isArray(tiposDeCarga) ? tiposDeCarga : [];
}
Trailer.prototype.puedeCargar = function (tipoDeCargamento) {
    return this.tiposDeCarga.includes(tipoDeCargamento);
};
Trailer.prototype.cargarTrailer = function (carga) {
    return 'Trailer esta cargado con ' + carga;
};

function CamionLigero(trailer, tipoDeCamion) {
    Trailer.call(this);
    this.ejes = 2;
    this.trailer = trailer;
    this.tipoDeCamion = tipoDeCamion;
}
CamionLigero.prototype = Object.create(Trailer.prototype);
CamionLigero.constructor = CamionLigero;
CamionLigero.prototype.puedeCargar = function (tipoDeCargamento) {
    return this.trailer.puedeCargar(tipoDeCargamento);
    console.log('Tipo de cargamento: ' + tipoDeCargamento);
};

function Conductor(nombre, camion) {
    this.nombre = nombre;
    this.camion = camion;
}
Conductor.prototype.puedeCargar = function (tipoDeCargamento) {

};
